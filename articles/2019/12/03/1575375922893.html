最小生成树模板

prim 算法 按点算，适合于点少的情况

邻接矩阵，时间复杂度 o(n^2)

 1//PTA公路村村通 
 2#include<iostream>
 3#include<cstdio>
 4using namespace std;
 5const int INF = 0x3fffffff;
 6const int MAXV = 1005;
 7int G[MAXV][MAXV], d[MAXV];//G存两个城镇直接的距离，d存已标记点到此点的最短距离 
 8bool vis[MAXV];//vis存是否被标记过 
 9int prim(int n)
10{
11	int ans = 0;//边权最小值 
12	fill(vis,vis+MAXV,false);
13	fill(d, d + MAXV, INF);
14	d[1] = 0;//从第一个城镇开始 
15	for (int i = 0; i < n; i++)//n个点，循环n次 
16	{
17		int u = -1, Min = INF;//Min存放最小d[u],u点的d[u]最小 
18		for (int j = 1; j <= n; j++)//寻找未访问点中最小d[] 
19		{
20			if (vis[j] == false && d[j] < Min)
21			{
22				u = j;
23				Min = d[j];
24			}
25		}
26		if (u == -1)return -1;//不连续，无法连成一棵树 
27		vis[u] = true;//标记已访问 
28		ans += d[u];
29		for (int k = 1; k <= n; k++)//更新与u点连接的最优值 
30		{
31			//k未访问&&k能到达u&&u为中介点使k离集合更短 
32			if (vis[k] == false && G[u][k] != INF&&G[u][k] < d[k])
33				d[k] = G[u][k];
34		}
35	}
36	return ans;
37}
38int main()
39{
40	int n, m, i, a, b, c, ans;//n代表城镇个数，m代表路径个数 
41	while(scanf("%d %d",&n,&m)!=EOF&&n)
42	{
43	    fill(G[0], G[0] + MAXV*MAXV, INF);//G初始化为没有任何路径连通 
44	for (i = 0; i < m; i++)
45	{
46		cin >> a >> b >> c;//输入两个城镇与权值 
47		G[a][b] = G[b][a] = c;
48	}
49	ans = prim(n);
50	cout << ans << endl;//ans存最后结果 
51	}
52	return 0;
53}

邻接表法，时间复杂度 o(nlogn+e)

 1//PTA公路村村通 
 2#include<iostream>
 3#include<cstdio>
 4#include<vector> 
 5using namespace std;
 6const int INF = 0x3fffffff;
 7const int MAXV = 1005;
 8struct node
 9{
10	int v,dis;
11};//v为边的目标顶点，dis为边权 
12vector<node> s[MAXV];
13int d[MAXV];//d存已标记点到此点的最短距离 
14bool vis[MAXV];//vis存是否被标记过 
15int prime(int n)
16{
17	int ans = 0;//边权最小值 
18	fill(vis,vis+MAXV,false);
19	fill(d, d + MAXV, INF);
20	d[1] = 0;//从第一个城镇开始 
21	for (int i = 0; i < n; i++)//n个点，循环n次 
22	{
23		int u = -1, Min = INF;//Min存放最小d[u],u点的d[u]最小 
24		for (int j = 1; j <= n; j++)//寻找未访问点中最小d[] 
25		{
26			if (vis[j] == false && d[j] < Min)
27			{
28				u = j;
29				Min = d[j];
30			}
31		}
32		if (u == -1)return -1;//不连续，无法连成一棵树 
33		vis[u] = true;//标记已访问 
34		ans += d[u];
35		for (int k = 0; k < s[u].size(); k++)//更新与u点连接的最优值 
36		{
37			int v=s[u][k].v;//通过邻接表直接获得u能到达的顶点v 
38			//k未访问&&u为中介点使k离集合更短 
39			if (vis[v] == false && s[u][k].dis < d[v])
40				d[v] = s[u][k].dis;
41		}
42	}
43	return ans;
44}
45int main()
46{
47	int n, m, i, a, b, c, ans;//n代表城镇个数，m代表路径个数 
48	while(scanf("%d %d",&n,&m)!=EOF&&n)
49	{
50		for(i=0;i<MAXV;i++)//清空所有邻接表 
51		{
52			s[i].clear();
53		}
54	for (i = 0; i < m; i++)
55	{
56		node t;
57		cin >> a >> b >> c;//输入两个城镇与权值 
58		t.v=b;
59		t.dis=c;
60		s[a].push_back(t);
61		t.v=a;
62		t.dis=c;
63		s[b].push_back(t);
64	}
65	ans = prime(n);
66	cout << ans << endl;//ans存最后结果 
67	}
68	return 0;
69}

kruskal 算法 适用于边少的情况

 1#include<iostream>
 2#include<algorithm>
 3#include<cstdio>
 4using namespace std;
 5const int MAXN=1005; 
 6const int N=105; 
 7struct node
 8{
 9	int x, y, z;//x,y代表端点，z权值 
10}s[MAXN];
11int father[N];//并查集数组 
12int findfather(int v)//并查集函数 
13{
14	if (v == father[v])return v;
15	else 
16	{
17		int f = findfather(father[v]);
18		father[v]=f;//压缩路径 
19		return f;
20	}
21}
22bool cmp(node a, node b)//权值小的放前面 
23{
24	return a.z < b.z;
25}
26int kruskal(int n,int m)//n个顶点，m边数 
27{
28	int i,num=0,a,b,ans=0;//num存选择的边数，ans存最小权值 
29	for (i = 1; i <= n; i++)//并查集初始化 
30		father[i] = i;
31	sort(s + 1, s + m + 1, cmp);//按边的权值从小到大排序 
32	for (i = 1; i <= m; i++)
33		{
34			a = findfather(s[i].x); b = findfather(s[i].y);
35			if(a!=b)//不在一个集合 
36			{
37				father[b] = a;//合并集合 
38				num++;//道路数量相加 
39				ans += s[i].z;//权值 
40				if (num == n - 1)break;//边数等于顶点减 1时结束 
41			}
42		}
43		if(num!=n-1)return -1;//无法连通时返回-1 
44		else return ans;
45}
46int main()
47{
48	int m, n, i, sum;
49	while (scanf("%d %d", &n,&m)!=EOF && n && m)//n代表城镇个数，m代表路的个数 
50	{
51		for (i = 1; i <= m; i++)
52		{
53			scanf("%d %d %d", &s[i].x, &s[i].y, &s[i].z);
54		}
55		sum=kruskal(n,m);//sum为最小权值 
56		printf("%d\n", sum);
57	}
58	return 0;
59}
